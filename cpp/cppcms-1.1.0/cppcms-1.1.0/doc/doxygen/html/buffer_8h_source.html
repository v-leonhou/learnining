187  {
188  Buffer res;
189  typename Buffer::buffer_data_type data=buf.get();
190  while(data.second > 0 && n > 0) {
191  if(data.first->size <= n) {
192  n-= data.first->size;
193  data.second--;
194  data.first++;
195  }
